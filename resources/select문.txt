SELECT p.asset_Class, p.asset_Id, p.asset_Name, p.creation_Date, p.description, p.product, p.provider, p.provider_ID, p.verb,
p.provider_Content_Tier, p.metadata_Spec_Version, 
t.description, t.asset_Id, t.asset_Class, t.type, t.title, t.title_Brief, t.category, t.rating, t.summary_Short, t.run_Time,
t.display_Run_Time, t.provider_QA_Contact, t.billing_ID, t.licensing_Window_Start, t.licensing_Window_End, t.preview_Period,
t.title_Sort_Name, t.episode_Name, t.episode_Id, t.summary_Medium, t.summary_Long, t.actors_Display, t.chapter, t.studio_Name,
t.studio, t.closed_Captioning, t.season_Premiere, t.season_Finale, t.year, t.country_of_Origin, t.Genre, t.maximum_Viewing_Length,
t.suggested_Price, t.propagation_Priority, t.longTail_YN, t.director, t.writer_Display, t.contract_Name, t.distributor_Name,
m.description, m.asset_Id, m.asset_Class, m.type, m.encryption, m.audio_Type, m.screen_Format, m.dubbed_Languages, m.copy_Protection,
m.copy_Protection_Verbose, m.analog_Protection_System, m.encryption_Mode_Indicator, m.constrained_Image_Trigger, m.cgms_A, m.HDContent,
m.content_FileSize, m.content_CheckSum, m.value,
s.description, s.asset_Id, s.asset_Class, s.type, s.content_FileSize, s.content_CheckSum, s.value,
r.publication_Right, v.advisories
FROM package_table p INNER JOIN title_table t ON p.package_Id = t.package_Id
INNER JOIN movie_table m ON p.package_Id = m.package_Id
INNER JOIN poster_table s ON p.package_Id = s.package_Id
INNER JOIN publication_table r ON p.package_Id = r.package_Id
INNER JOIN advisories_table v ON p.package_Id = v.package_Id;
